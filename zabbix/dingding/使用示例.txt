	shell脚本版

	go二进制版

		cp config.yaml /root/.dingtalk/config.yaml
		chmod +x ./dingtalk
		cp ./dingtalk /usr/local/bin
		dingtalk text -c "test" -m '1383458xxxx,1375488xxxx'
